//PARA TROCAR MEU NOME
 git config --global user.name "kaleniuk2"

// PARA DEFINIR EMAIL
 git config --global user.email "guilhermekaleniuk@hotmail.com"

 //PARA DEFINIR O EDITOR QUE ESTOU USANDO NO MOMENTO (NÃO MUITO COMUM)
  git config --global core.editor sub

 //PARA VER QUAL USUARIO ESTÁ CADASTRADO NA MAQUINA

 git config user.name

 //PARA VER O EMAIL QUE ESTÁ CADASTRADO
 git config user.email

 //PARA VER TUDO Q ESTÁ CADASTRADO

 git config --list

 // PARA CRIAR UM REPOSITORIO

 git init


 //VER OQ FOI ADICIONADO, OQ FOI ALTERADO ETC, QUANDO TIVER EM VERMELHO É PQ O GIT NUNCA VIU

 git status

 // PAARA ADICIONAR NO GIT, E ELE PASSAR A MONITORAR OQ FOI ALTERADO 

 git add -A

 //PARA FAZER O COMMIT

 git commit -m "Primeiro commit dahora"

 //VER TODOS COMMITS FEITOS DAQUELE REPOSITORIO

 git log

 //PARA SABER EM QUAL BRENCH EU TO (TIPO VERSAO MASTER, 2.0, ETC );

 git branch

 //COMMITAR SEM TER Q PASSAR PELO "GIT ADD -A"

 git commit -am "Terceiro commit muito bom"

 //VOLTAR AO Q TAVA ANTES, (SOFT = VOLTAR AO Q ESTAVA ANTES POREM COM ACESSO AOS ARQUIVOS... MANTER OS ARQUIVOS E VOLTAR COMO ESTAVA ANTES DE DAR O COMMIT E O ADD -A, HARD = VOLTAR E APAGAR TUDO)

 git reset --hard (id do commit, vejo em git log)

 // CRIAR NOVO BRANCH

 git branch test


 // TROCAR DE BRANCH 

 git checkout test
