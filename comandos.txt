 git config --global user.name "kaleniuk2"

 git config --global user.email "guilhermekaleniuk@hotmail.com"

 //PARA DEFINIR O EDITOR QUE ESTOU USANDO NO MOMENTO (NÃO MUITO COMUM)
  git config --global core.editor sub

 //PARA VER QUAL USUARIO ESTÁ CADASTRADO NA MAQUINA

 git config user.name

 //PARA VER O EMAIL QUE ESTÁ CADASTRADO
 git config user.email

 //PARA VER TUDO Q ESTÁ CADASTRADO

 git config --list

 // PARA CRIAR UM REPOSITORIO

 git init


 //VER OQ FOI ADICIONADO, OQ FOI ALTERADO ETC, QUANDO TIVER EM VERMELHO É PQ O GIT NUNCA VIU

 git status

 // PAARA ADICIONAR NO GIT, E ELE PASSAR A MONITORAR OQ FOI ALTERADO 

 git add -A

 //PARA FAZER O COMMIT

 git commit -m "Primeiro commit dahora"

 //VER TODOS COMMITS FEITOS DAQUELE REPOSITORIO

 git log

 //PARA SABER EM QUAL BRENCH EU TO (TIPO VERSAO MASTER, 2.0, ETC );

 git branch

 //COMMITAR SEM TER Q PASSAR PELO "GIT ADD -A"

 git commit -am "Terceiro commit muito bom"

 //VOLTAR AO Q TAVA ANTES, (SOFT = VOLTAR AO Q ESTAVA ANTES POREM COM ACESSO AOS ARQUIVOS... MANTER OS ARQUIVOS E VOLTAR COMO ESTAVA ANTES DE DAR O COMMIT E O ADD -A, HARD = VOLTAR E APAGAR TUDO)

 git reset --hard (id do commit, vejo em git log)

 // CRIAR NOVO BRANCH

 git branch test


 // TROCAR DE BRANCH 

 git checkout test


// PARA VER QUAIS LINHAS FORAM MUDADAS NOS ARQUIVOS

git diff

//VER NOME DOS ARQUIVOS Q FORAM ALTERADOS
git diff --name-only

//SABER OQ FOI ALTERADO SOMENTE NAQUELE ARQUIVO

git diff style.css (por exemplo)

//TIRAR MODIFICACOES QUE FORAM FEITAS, E ESTAO NA FILA PRA DAR COMMIT

git checkout HEAD -- style.css (por exemplo)


//COMANDO PARA CADASTRAR A KEY NO PC, PARA CONSEGUIR FAZER ALTERAÇOES NOS REPOSITORIOS DA CONTA DO GIT
ssh-keygen -t rsa -b 4096 -C "your_email@example.com"

//COMANDO PARA VINCULAR O REPOSITORIO LOCAL COM REPOSITORIO REMOTO
git remote add origin https://github.com/kaleniuk2/github_bonieky.git

//COMANDO PARA VER SE O REPOSITORIO FOI MESMO VINCULADO
git remote // E PARA VER COM MAIS DETALHES git remove -v

// COMANDO PARA COLOCAR O REPOSITORIO LOCAL NA NUVEM
git push -u origin master


//COMANDO PARA FAZER AS ALTERAÇOES DO LOCAL PARA O REMOTO (PRIMEIRO FAZ O COMMIT NO LOCAL git commit -am "BLA BLA")

git push origin master
